## This file contains events that interact with the vanilla precursor
## events to increase players' opportunities to complete them. A
## player who starts a precursor chain will generally be able to
## eventually complete it, although it may take a while, especially
## with a small empire.

## These events are numbered with the number of the anomaly event that
## they are mimicking. So for example mem_precursor_amendment.5
## is used to spawn precursor_events.5's project on an owned planet.

namespace = mem_precursor_amendment

### The Vultaum Star Assembly

planet_event = {
	id= mem_precursor_amendment.5
	title = mem_precursor_amendment.generic.title
	desc = mem_precursor_amendment.generic.desc
	picture = GFX_evt_archaeological_dig
	
	trigger = {
		is_capital = no
		exists = owner
		owner = { 
			has_country_flag = vultaum_intro
			NOT = { 
				has_country_flag = vultaum_system 
				has_country_flag = vultaum_1
			} 
		}
	}
	
	mean_time_to_happen = {
		months = 3600
		modifier = {
			factor = 10
			years_passed < 10
		}
		modifier = {
			factor = 8
			years_passed < 20
		}
		modifier = {
			factor = 6
			years_passed < 25
		}
		modifier = {
			factor = 4
			years_passed < 35
		}
		modifier = {
			factor = 3
			years_passed < 50
		}
		modifier = {
			factor = 2
			years_passed < 75
		}
		modifier = {
			factor = 0.8
			years_passed > 100
		}
		modifier = {
			factor = 0.6
			years_passed > 150
		}
	}
	
	immediate = {
		hidden_effect = {
			owner = { set_country_flag = vultaum_1 }
		}
	}
	
	option = {
		name = FASCINATING
		enable_special_project = {
			name = "VULTAUM_ARCHEOLOGY_1"
			location = this
			owner = root.owner
		}
	}
}
